Example #4 - Types of pickers

regular

Usage:

var picker = new smartPicker("select", "my-list", {
		contentData: {
			text: "Select one",
			options: [{ value: "1", name: "Select this"},
				  { value: "2", name: "Or this!"},
				  { value: "3", name: "How about this one?"}]
		}
	});
picker.show();

mult

Selected Countries:

Usage:

var picker = new smartPicker("mult", "my-mult", {
	width: 500,
	draggable: true,
	contentData: {
		text: "Select one or many",
		options: [{ value: "1", name: "Select this"},
			  { value: "2", name: "Select this as well if you want"},
			  { value: "3", name: "What about this one?"}]
	},
	confirmButton: "Select these",
	onInteraction: function(data) {
		var selected = data.selected;
		var selected_text = "";
		for (var i = 0, size=selected.length; i < size; i++) {
			selected_text += selected[i].text;
			if(i < size - 1) selected_text+= ", ";
		};
		$("#result").text(selected_text);
	},
	onSet: function(data) {
		var selected = data.selected;
		var selected_text = "";
		for (var i = 0, size=selected.length; i < size; i++) {
			selected_text += selected[i].text;
			if(i < size - 1) selected_text+= ", ";
		};
		alert("You selected: "+selected_text);
	}
});
picker.show();
